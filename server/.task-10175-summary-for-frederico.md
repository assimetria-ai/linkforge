## Quick Usage Example

```javascript
const { createCrudRouter } = require('../../lib/@system/Helpers')

// One line creates complete REST API with pagination, search, sorting, filtering
module.exports = createCrudRouter({
  repo: ProductRepo,
  config: { basePath: '/api/products' }
})

// Automatically creates:
// GET    /api/products       - List (paginated, searchable, sortable, filterable)
// GET    /api/products/:id   - Get by ID
// POST   /api/products       - Create
// PATCH  /api/products/:id   - Update
// DELETE /api/products/:id   - Delete
```
