## Usage Examples

### Example 1: Auto-Generated CRUD API (Zero Boilerplate)

```javascript
const { createCrudRouter } = require('../../lib/@system/Helpers')
const ProductRepo = require('../../db/repos/ProductRepo')
const { CreateProductSchema, UpdateProductSchema } = require('./schemas')

// One configuration object creates 5 complete REST endpoints
module.exports = createCrudRouter({
  repo: ProductRepo,
  validation: {
    create: CreateProductSchema,
    update: UpdateProductSchema,
  },
  config: {
    basePath: '/api/products',
    dataKey: 'product',
  },
})

// Automatically creates:
// GET    /api/products       - List with pagination, search, filtering, sorting
// GET    /api/products/:id   - Get by ID
// POST   /api/products       - Create with validation
// PATCH  /api/products/:id   - Update with validation
// DELETE /api/products/:id   - Delete
```

### Example 2: Manual API with All Middleware

```javascript
const express = require('express')
const router = express.Router()
const {
  pagination,
  sorting,
  filtering,
  handleList,
  asyncHandler,
} = require('../../lib/@system/Helpers')

router.get('/api/products',
  pagination({ defaultLimit: 20, maxLimit: 100 }),
  sorting({ 
    allowedFields: ['name', 'price', 'created_at'],
    defaultField: 'created_at',
    defaultOrder: 'desc',
  }),
  filtering({
    allowedFields: ['category', 'in_stock', 'brand'],
    booleanFields: ['in_stock'],
  }),
  asyncHandler(async (req, res, next) => {
    await handleList({
      repo: ProductRepo,
      req, res, next,
      filters: {
        whereClause: req.filters.whereClause,
        params: req.filters.params,
        orderBy: req.sorting.clause,
      },
    })
  })
)

module.exports = router
```

**Query examples:**
- `/api/products?page=2&limit=20`
- `/api/products?sort=price:asc&category=electronics&in_stock=true`
- `/api/products?price[gte]=100&price[lte]=500&sort=price:asc`

### Example 3: Search with Full-Text

```javascript
const { buildSearchCondition, parseSearchQuery } = require('../../lib/@system/Helpers')

router.get('/api/products/search', 
  pagination(),
  asyncHandler(async (req, res) => {
    const searchQuery = parseSearchQuery(req.query.q)
    const searchCondition = buildSearchCondition({
      query: searchQuery,
      fields: ['name', 'description', 'brand'],
      mode: 'contains', // or 'starts_with', 'exact'
    })
    
    const products = await ProductRepo.search({
      ...searchCondition,
      limit: req.pagination.limit,
      offset: req.pagination.offset,
```
