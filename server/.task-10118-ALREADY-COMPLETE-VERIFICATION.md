## 🎓 Developer Quick Start

For developers wanting to use this scaffolding:

```javascript
// 1. Import helpers
const {
  pagination,
  sorting,
  filtering,
  handleList,
  parseSearchQuery,
  buildWhereClause,
} = require('./lib/@system/Helpers')

// 2. Create endpoint
router.get('/api/products',
  pagination(),
  sorting({ allowedFields: ['name', 'price'] }),
  filtering({ allowedFields: ['category'] }),
  async (req, res, next) => {
    const search = parseSearchQuery(req.query, {
      defaultFields: ['name', 'description'],
    })
    
    const { whereClause, params } = buildWhereClause({
      searchQuery: search.query,
      searchFields: search.fields,
      filters: req.filters.parsed,
    })
    
    await handleList({
      repo: ProductRepo,
      req, res, next,
      filters: {
        whereClause,
        params,
        orderBy: req.sorting.clause,
      },
    })
  }
)
```

```javascript
const { createCrudRouter } = require('./lib/@system/Helpers')

  repo: ProductRepo,

})
```
